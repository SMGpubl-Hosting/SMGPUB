<script>
const API = "/api";

async function loadProducts() {
    try {
        const res = await fetch(API + "/products");
        const data = await res.json();
        
        document.getElementById("products").innerHTML = data.map(p => `
            <div class="card">
                <span class="tag">${p.type === 'preset' ? 'Logic & Mixing' : 'Trap Instrumental'}</span>
                <h3>${p.id}</h3>
                <p style="font-size: 20px; margin: 10px 0; color: var(--accent);">$${p.price}</p>
                <button class="btn" style="width:100%" onclick="buy('${p.id}')">
                    ${p.type === 'preset' ? 'Download Preset' : 'Instant Lease'}
                </button>
                <a href="https://www.beatstars.com/smgpub" target="_blank" class="market-link">
                    View on BeatStars Marketplace
                </a>
            </div>
        `).join('');
    } catch (err) {
        console.error("Store Load Error:", err);
    }
}

async function buy(id) {
    const res = await fetch(API + "/stripe/create-checkout", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ 
            productId: id,
            // Include source for tracking
            source: "smgpub_direct" 
        })
    });
    const data = await res.json();
    if(data.url) window.location.href = data.url;
}

// Initial Load
loadProducts();
</script>
